\begin{frame}{Modeling Wells Fargo: Destructive Effort}

\begin{wideitemize}
    \item Output depends on a productive and destructive task:
    \[y = 1+  ae_1 -b e_2, a>0, b>0\]
    \item The $1$ is just profit from operating at all.
    \item The cost of effort is:
    \[c(e_1, e_2) = (e_1^2+e_2^2)/2 \]
      \item We measure and pay only based on total effort:
    \[m=e_1+e_2\]
    \[w(m)=\alpha + \beta m\]
\end{wideitemize}
\end{frame}

\begin{frame}{Modeling Wells Fargo: Destructive Effort}
\begin{theorem}
    If marginal destruction outweighs marginal production ($a-b>0$) the optimal wage is $\beta^*=(a-b)/2$. If not, the firm pays a flat salary: $\beta^*=0$.
\end{theorem}
\begin{wideitemize}
    \item Incentives are not always optimal.
    \item Sometimes a flat salary is more profitable.
    \item How does this relate to gaming?
    \item How is it different?
    \item If the firm can prohibit $e_2$, everything is ``resolved."
\end{wideitemize}
\end{frame}
